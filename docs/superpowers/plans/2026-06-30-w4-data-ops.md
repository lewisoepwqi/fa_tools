```python
from alembic.autogenerate import compare_metadata
from alembic.migration import MigrationContext
from sqlalchemy import create_engine
from app.db.base import Base


def test_no_autogenerate_diff(tmp_path, monkeypatch):
    """upgrade 到 head 后，Base.metadata 与迁移产出的 schema 必须零 diff。
    钉死 conftest 的 create_all 与迁移两条 schema 路径，防止改模型不补迁移。"""
    db_file = tmp_path / "contract.db"
    url = f"sqlite+pysqlite:///{db_file}"
    monkeypatch.setenv("DATABASE_URL", url)
    from app.core.config import get_settings
    get_settings.cache_clear()

    alembic_cfg = _make_alembic_config()  # 复用文件内现有 helper；若名字不同按实际
    from alembic import command
    command.upgrade(alembic_cfg, "head")

    engine = create_engine(url)
    with engine.connect() as conn:
        ctx = MigrationContext.configure(
            conn, opts={"compare_type": True, "target_metadata": Base.metadata}
        )
        diffs = compare_metadata(ctx, Base.metadata)
    engine.dispose()
    get_settings.cache_clear()
    assert diffs == [], f"迁移与 Base.metadata 不一致: {diffs}"
```

> 若文件无 `_make_alembic_config` helper，参照现有 `test_initial_migration_matches_contract` 里构造 `Config` 的写法（`Config()` + `set_main_option("script_location", ...)` + url）。`target_metadata` 放进 opts 让类型比较更准。
